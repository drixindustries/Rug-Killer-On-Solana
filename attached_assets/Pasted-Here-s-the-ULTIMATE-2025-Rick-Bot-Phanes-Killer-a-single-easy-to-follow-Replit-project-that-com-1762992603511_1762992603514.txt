def get_birdeye(ca):
    url = f"{BIRDEYE}/defi/overview?address={ca}"
    r = requests.get(url, headers=headers_birdeye)
    return r.json().get("data", {}) if r.ok else {}

def get_price_history(ca):
    url = f"{BIRDEYE}/defi/price_history?address={ca}&time_from={int(time.time()-86400)}"
    r = requests.get(url, headers=headers_birdeye)
    return r.json().get("data", []) if r.ok else []

def get_holders(ca):
    url = f"{BIRDEYE}/token/top_holders?address={ca}"
    r = requests.get(url, headers=headers_birdeye)
    return r.json().get("data", [])[:20] if r.ok else []

def get_rugcheck(ca):
    r = requests.get(f"{RUGCHECK}/{ca}/report", headers={"Authorization": os.getenv("RUGCHECK_KEY", "")})
    return r.json() if r.ok else {"risk_score": "API error"}

def get_helius_metadata(ca):
    r = requests.post(HELIUS, json={"mintAccounts": [ca]})
    data = r.json()
    return data[0] if data else {}

def check_pumpfun(ca):
    # Detect if launched on pump.fun
    url = f"https://pump.fun/api/token/{ca}"
    try:
        r = requests.get(url)
        if r.ok:
            info = r.json()
            return {"pumpfun": True, "dev_bought": info.get("dev_bought", 0), "bonding_curve": info.get("progress", 0)}
    except: pass
    return {"pumpfun": False}

def detect_snipers(holders):
    bundled = 0
    for h in holders[:10]:
        if h.get("tag") == "Bundle" or "sniper" in h.get("owner", "").lower():
            bundled += 1
    return bundled
